## Introduction to @jkt48/core Module
The `@jkt48/core` module is an official NPM package that provides a simplified wrapper for JKT48Connect API, making it easier to integrate JKT48 data into your JavaScript and Node.js applications. This module handles authentication, error management, and provides a clean interface for all available endpoints.

## Error Handling Best Practices

<CodeBlock
    marginBottom="16"
    codes={[
  {
    code: `import jkt48Api from '@jkt48/core';

async function robustApiCall() {
  const apiKey = 'your-api-key-here';
  
  try {
    // Implement retry logic for better reliability
    const maxRetries = 3;
    let attempt = 0;
    
    while (attempt < maxRetries) {
      try {
        const members = await jkt48Api.members(apiKey);
        return members; // Success
      } catch (error) {
        attempt++;
        
        if (attempt === maxRetries) {
          throw error; // Final attempt failed
        }
        
        // Wait before retry (exponential backoff)
        await new Promise(resolve => 
          setTimeout(resolve, Math.pow(2, attempt) * 1000)
        );
      }
    }
  } catch (error) {
    // Handle different error types
    if (error.status === 429) {
      console.error('Rate limit exceeded. Please try again later.');
    } else if (error.status === 401) {
      console.error('Invalid API key. Please check your credentials.');
    } else if (error.status === 500) {
      console.error('Server error. Please try again later.');
    } else {
      console.error('Unexpected error:', error.message);
    }
    
    throw error;
  }
}`,
    language: "javascript",
    label: "Error Handling Example"
  }
]} />

## React Integration Example

<CodeBlock
    marginBottom="16"
    codes={[
  {
    code: `import React, { useState, useEffect } from 'react';
import jkt48Api from '@jkt48/core';

function JKT48MemberList() {
  const [members, setMembers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    async function fetchMembers() {
      try {
        const apiKey = process.env.REACT_APP_JKT48_API_KEY;
        const memberData = await jkt48Api.members(apiKey);
        setMembers(memberData);
      } catch (err) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    }

    fetchMembers();
  }, []);

  if (loading) return <div>Loading JKT48 members...</div>;
  if (error) return <div>Error: {error}</div>;

  return (
    <div>
      <h2>JKT48 Members ({members.length})</h2>
      <ul>
        {members.map(member => (
          <li key={member.id}>
            {member.name} - {member.generation}
          </li>
        ))}
      </ul>
    </div>
  );
}

export default JKT48MemberList;`,
    language: "javascript",
    label: "React Component Example"
  }
]} />
